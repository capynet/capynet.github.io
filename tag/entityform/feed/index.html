< ? //Vamos a crear la entidad con el usuario que esté logueado. global $user; //solicitud_de_informacion es el Entityform type (admin/structure/entityform_types) $entityform = entity_create('entityform', array( 'type' => "solicitud_de_informacion",
  'created' => time(),
  'changed' => time(),
  'language' => LANGUAGE_NONE,
  'uid' => $user->uid
));

//Nos simplificamos la vida usando Entity API
$wrapper = entity_metadata_wrapper('entityform', $entityform);

//Rellenamos los fields
$wrapper->field_nombre = "Marcelo";
$wrapper->field_apellidos = "Tosco";
$wrapper->field_email = "marcelo@tosco.com";
$wrapper->field_usted_dira = "digo yo";

//Done!
$wrapper->save();
