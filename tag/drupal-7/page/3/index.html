  'created' => time(),
  'changed' => time(),
  'language' => LANGUAGE_NONE,
  'uid' => $user->uid

//Nos simplificamos la vida usando Entity API
$wrapper = entity_metadata_wrapper('entityform', $entityform);
 
//Rellenamos los fields
$wrapper->field_nombre = "Marcelo";

$wrapper->field_email = "marcelo@tosco.com";
$wrapper->field_usted_dira = "digo yo";
 
//Done!
$wrapper->save();

function MIMODULO_form_alter(&$form, &$form_state, $form_id) {
  if (in_array($form_id, array('user_login', 'user_login_block'))) {
    $form['name']['#attributes']['placeholder'] = t('Username');
    $form['pass']['#attributes']['placeholder'] = t('Password');
    $form['name']['#title_display'] = "invisible";
    $form['pass']['#title_display'] = "invisible";
  }
