Linux: Convertir imágenes de png a jpg desde la consola

  • capynet
  • 13 May, 2013
  • Linux, Tip
  • Linux, Programas
  • 4 comments

Una vez mas esto es una nota mental mas que un tutorial: Si no tenés instalado imagemagick:

1
sudo apt-get install imagemagick

Luego para convertir usamos el comando convert:

1
convert imagen.png imagen.jpg

Y para convertir todas las imagenes en un directorio:

1
convert *.png *.jpg

Chau!

app.get('index', function (req, res) {
    var messages = req.flash();
    var render_options = {
        titulo: "Página principal"
    };
 
    if (messages.error || messages.info || messages.success) {
        if (messages.error) messages.hasError = messages.error.length > 0;
        if (messages.info) messages.hasInfo = messages.info.length > 0;
        if (messages.success) messages.hasSuccess = messages.success.length > 0;
        render_options.messages = messages;
    }
 
    res.render('index', render_options);
});
