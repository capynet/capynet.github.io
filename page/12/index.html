function download_file($archivo, $downloadfilename = null) {
    if (file_exists($archivo)) {
        $downloadfilename = $downloadfilename !== null ? $downloadfilename : basename($archivo);

        header('Content-Length: ' . filesize($archivo));
        ob_clean();
        flush();
        readfile($archivo);
